#include <SPI.h>

byte ssPin = 10;

byte DisplayArray[] = {
 0,0,0,0,0,0,0,0,0,                  // ALL_ON
 255,255,255,255,255,255,255,255,255 // ALL_OFF
};

void setup(){
 SPI.begin();
 }

void loop(){
   digitalWrite (ssPin, LOW);
   for (x=0; x<9; x=x+1){
       SPI.transfer (DisplayArray[x]);
       }
   digitalWrite (ssPin, HIGH);

   delay (750);

   digitalWrite (ssPin, LOW);
   for (x=0; x<9; x=x+1){
       SPI.transfer (DisplayArray[x]);
       }
   digitalWrite (ssPin, HIGH);

   delay (750);  
}
